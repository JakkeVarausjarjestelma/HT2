CREATE TABLE IF NOT EXISTS "Seura" (
"SeuraID" INTEGER PRIMARY KEY,
"Nimi" VARCHAR(32));
CREATE TABLE IF NOT EXISTS "Lajivalinta" (
"LajiID" INTEGER PRIMARY KEY,
"Nimi" VARCHAR(32)
);
CREATE TABLE IF NOT EXISTS "Sali" (
"SaliID" INTEGER PRIMARY KEY,
"LajiID" INTEGER,
"Nimi" VARCHAR(32),
FOREIGN KEY("LajiID") REFERENCES "Laji"("LajiID"));
CREATE TABLE IF NOT EXISTS "Valineet" (
"ValineID" INTEGER PRIMARY KEY,
"SaliID" INTEGER,
"Nimi" VARCHAR(32),
FOREIGN KEY("SaliID") REFERENCES "Sali"("SaliID"));
CREATE TABLE IF NOT EXISTS "Varaus" (
"VarausID" INTEGER PRIMARY KEY,
"VaraajaID" INTEGER,
"SaliID" INTEGER,
"Aloitusaika" TIME,
"Kesto" INTEGER,
"Pvm" DATE,
FOREIGN KEY("VaraajaID") REFERENCES "Varaaja"("VaraajaID")
FOREIGN KEY("SaliID") REFERENCES "Lajivalinta"("SaliID")
);
CREATE TABLE IF NOT EXISTS "Henkilo" (
"Nimi" VARCHAR(32),
"Puhelinnumero" VARCHAR(16),
"VaraajaID" INTEGER PRIMARY KEY,
FOREIGN KEY("VaraajaID") REFERENCES "Varaaja"("VaraajaID")
);
CREATE TABLE IF NOT EXISTS "Varaaja" (
"VaraajaID" INTEGER PRIMARY KEY,
"SeuraID" INTEGER,
FOREIGN KEY("SeuraID") REFERENCES "Seura"("SeuraID")
);
